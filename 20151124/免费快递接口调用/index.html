免费快递接口调用

在电商系统中，查看某个订单的详细信息，展示订单的物流跟踪记录已是标配。

以下是通过jQuery调用快递100的url获取结果，达到查询订单跟踪的目的。

var com = "yuantong"; //快递公司名，如圆通为yuantong, 京东为jd
var orderNO = "880589222679907360";  //订单号
//获取快递信息
$.ajax({
    url: "http://www.kuaidi100.com/query?type="+com+"&postid="+orderNO+"&id=1&valicode=&temp="+Math.random(),
    dataType: 'jsonp',

    success: function (result) {
        if(result.message == "ok"){

            //展示内容，根据情况自行调整
            var strItem = "";
            var today = new Array('周日','周一','周二','周三','周四','周五','周六');  

            for(i = result.data.length-1;i>=0;i--){
                if(i==0){
                    strItem += "<li class=\"clearfix current\">";
                }
                else{
                    strItem += "<li class=\"clearfix\">";
                }
                strItem += "<span class=\"day\">"+result.data[i].time.replace(/([\d-]+)[\s]+[\d:]+/g,'$1')
                    +today[new Date(result.data[i].time.replace(/([\d]+)-([\d]+)-([\d]+)[\s]+[\d:]+/g,'$2-$3-$1')).getDay()]+"</span>";
                strItem += "<span class=\"tg time\">"+result.data[i].time.replace(/[\d-]+[\s]+([\d:]+)/g,'$1')+"</span>";
                strItem += "<span class=\"tg things\">"+result.data[i].context+"</span>";
            }
            $(".ddzzl ul").append(strItem);
            //展示内容，根据情况自行调整
        }
    },
    error:function(){
        alert("error");
    }

});
